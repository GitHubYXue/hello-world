function formatTime(timestamp) {
var date = new Date(timestamp * 1000);
Y = date.getFullYear();
M = (date.getMonth() + 1 < 10 ? '0' + (date.getMonth() + 1) : date.getMonth() + 1);
D = date.getDate();
h = date.getHours();
m = date.getMinutes() < 10 ? '0' + date.getMinutes() : date.getMinutes();
s = date.getSeconds();

var time = new Object();
time.y = Y;
time.m = M;
time.d = D;
time.h = h;
time.i = m;
time.s = s;
return time; //呀麻碟
}
